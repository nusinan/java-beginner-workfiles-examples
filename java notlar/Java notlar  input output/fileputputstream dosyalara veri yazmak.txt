
package javaýo;

import java.io.File;
import java.io.FileNotFoundException;
import java.io.FileOutputStream;
import java.io.IOException;
import java.util.logging.Level;
import java.util.logging.Logger;


public class JavaIO {

 
    public static void main(String[] args) {
      
        //File file =new File(" dosya.txt"); burdan bir dosya olusturabilir ve assagidaki dosya.txt icerisine ekleyebiliriz
        FileOutputStream fos =null;
        try {    
            fos= new FileOutputStream("dosya.txt",true);  //olusturdugumuz dosyayý buraya eklemek yerine direk olarak bir txt dosyasý olsuturabiliriz
            try {                                // olusturdugumuz dosyayý her ccalistýrdýgýmýzda eski yazdýgý islem uzerinden ayný islemleri tekrar
                                                 //yapmasýný istiyorsak true ifadesini kullanmalýyýz kullanmadýgýmýz zaman  dosya islemleri her defasýnda silerek 
                                                  // var olan islemleri tekrar yaptýrtýr
                                                                                           
                fos.write(65);                  // dosya da yazdýrma islemi yapmak icin byte code seklinde yazdýrmak zorundayýz 
                byte [] Array ={101,108,77,68};  // byte arreyi
                fos.write(Array);
                String s ="numansinan";   // dosyaya yazdýrmak icin bir String olusturduk
                byte[] s_array=s.getBytes();  //yazdýgýmýz stringi olusturdugumuz bir byte arrey ine donusturduk
                fos.write(s_array);          // yazýmýzýn byte sekline donusmus halini yazdýrma methoduna gondererek ekrana yazdýrdýk
                
               
            } catch (IOException ex) {
                System.out.println("dosyaya yazýlýrken bir hata olustu...");  // program bir hata cýkabilecegini bildirdigi icin  try cath blogu olusturduk
            }
        } catch (FileNotFoundException ex) {
            System.out.println(" file not found exception olustu...");
        }
        
        
        finally{   // dosyayýnýn iþi bittigi zaman kapatma methodunu exception olusma ihtimaline karþý  close u iceride calýstýrmasýný bildirdik
        
            try {
                fos.close();             // close methodu cagrýldýgý zaman da  exception olusabilcegi icin  try cath ile yakaladýk
            } catch (IOException ex) {
                System.out.println(" dosya kapatýlýrken bir hata olustu...");
            }
        
        }
    }
    
}

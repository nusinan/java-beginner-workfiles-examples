import java.util.LinkedList;

import java.util.logging.Logger;

public class Main {
    public static void main(String[] args) {
        
        List<Integer> list =  new LinkedList<Integer>();
        
        Thread t = new Thread(new Runnable() {
            @Override
            public void run() {
                System.out.println("Thread çalýþýyor....");
                
                for (int i = 0 ; i < 1000000 ;i++){
                    
                    if (Thread.currentThread().isInterrupted()){    // calsýtýgý anda thread i interrupt eden bir sey varmý yok mu 
                        System.out.println("Kesintiye Uðradý....");//  onun kontrolunu yapýyor
                        return;
                        
                    }
                    
                    list.add(i);  // eger interrupt exception hatasý olmadýysa  linked list imize deger eklemesi yapýyor
                   
                }
                
                System.out.println("Thread kesintiye uðramadan iþini bitirdi...");
                
                
                /*for(int i= 1; i <= 10;i++) {

                    }
                    
                }*/
             
            }
        });
        
        t.start(); // thread in calsýmasýný bildiriyoruz 
        
        try {
            Thread.sleep(5000);  // 5 saniye bekletiyoruz  thread 5 sn de yukarýdaki for dongusundeki islemi bitiridigi icin if
        } catch (InterruptedException ex) {   // dongusunu atlayýp islemine devam ediyor ve interruptexception hatasý vermeden 
            Logger.getLogger(Main.class.getName()).log(Level.SEVERE, null, ex);// program sonlanýyor
        }
        t.interrupt();    // thread i yarýda kesme methodu ama thread  5 sn icinde  for dongusunu bitirdigi icin dongunun  icindeki 
                          // exception hatasýný gormuyor ve interrupt methodu calsýmýyor
     
    }
